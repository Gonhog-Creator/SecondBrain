services:
  # PostgreSQL Database
  postgres:
    image: postgres:15
    container_name: deltavision-postgres
    environment:
      POSTGRES_DB: deltavision
      POSTGRES_USER: deltavision_user
      POSTGRES_PASSWORD: deltavision_pass
      POSTGRES_INITDB_ARGS: "--encoding=UTF-8 --lc-collate=C --lc-ctype=C"
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./scripts/init-db.sql:/docker-entrypoint-initdb.d/init-db.sql
    ports:
      - "5432:5432"
    networks:
      - deltavision-network
    restart: unless-stopped
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U deltavision_user -d deltavision"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 60s

  # PgAdmin for Database Management
  pgadmin:
    image: dpage/pgadmin4:latest
    container_name: deltavision-pgadmin
    environment:
      PGADMIN_DEFAULT_EMAIL: admin@deltavision.com
      PGADMIN_DEFAULT_PASSWORD: admin123
      PGADMIN_CONFIG_SERVER_MODE: 'False'
    volumes:
      - pgadmin_data:/var/lib/pgadmin
      - ./configs/pgadmin-servers.json:/pgadmin4/servers.json
    ports:
      - "5050:80"
    networks:
      - deltavision-network
    restart: unless-stopped
    depends_on:
      - postgres

  # Redis for Caching (Optional)
  redis:
    image: redis:7-alpine
    container_name: deltavision-redis
    command: redis-server --appendonly yes --requirepass redis123
    volumes:
      - redis_data:/data
    ports:
      - "6379:6379"
    networks:
      - deltavision-network
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "redis-cli", "--raw", "incr", "ping"]
      interval: 30s
      timeout: 10s
      retries: 3

  # DeltaVision API Server
  api:
    build:
      context: .
      dockerfile: Dockerfile
      target: api
    container_name: deltavision-api
    environment:
      - DATABASE_HOST=postgres
      - DATABASE_PORT=5432
      - DATABASE_NAME=deltavision
      - DATABASE_USER=deltavision_user
      - DATABASE_PASSWORD=deltavision_pass
      - REDIS_HOST=redis
      - REDIS_PORT=6379
      - REDIS_PASSWORD=redis123
      - API_HOST=0.0.0.0
      - API_PORT=8000
      - LOG_LEVEL=INFO
    volumes:
      - ./data:/app/data
      - ./configs:/app/configs
      - ./logs:/app/logs
      - ../vids:/app/vids  # Mount video directory
    ports:
      - "8000:8000"
    networks:
      - deltavision-network
    restart: unless-stopped
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/health"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 60s

  # DeltaVision Worker for Background Processing
  worker:
    build:
      context: .
      dockerfile: Dockerfile
      target: worker
    environment:
      - DATABASE_HOST=postgres
      - DATABASE_PORT=5432
      - DATABASE_NAME=deltavision
      - DATABASE_USER=deltavision_user
      - DATABASE_PASSWORD=deltavision_pass
      - REDIS_HOST=redis
      - REDIS_PORT=6379
      - REDIS_PASSWORD=redis123
      - WORKER_CONCURRENCY=2
      - LOG_LEVEL=INFO
    volumes:
      - ./data:/app/data
      - ./configs:/app/configs
      - ./logs:/app/logs
      - ../vids:/app/vids  # Mount video directory
    networks:
      - deltavision-network
    restart: unless-stopped
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy

  # Nginx Reverse Proxy
  nginx:
    image: nginx:alpine
    container_name: deltavision-nginx
    volumes:
      - ./configs/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./data/exports:/var/www/exports:ro  # Serve exported files
    ports:
      - "80:80"
      - "443:443"
    networks:
      - deltavision-network
    restart: unless-stopped
    depends_on:
      - api

  # Grafana for Visualization
  grafana:
    image: grafana/grafana:latest
    container_name: deltavision-grafana
    environment:
      - GF_SECURITY_ADMIN_PASSWORD=admin123
      - GF_USERS_ALLOW_SIGN_UP=false
      - GF_INSTALL_PLUGINS=grafana-clock-panel,grafana-simple-json-datasource
    volumes:
      - grafana_data:/var/lib/grafana
      - ./configs/grafana/provisioning:/etc/grafana/provisioning
      - ./configs/grafana/dashboards:/var/lib/grafana/dashboards
    ports:
      - "3000:3000"
    networks:
      - deltavision-network
    restart: unless-stopped
    depends_on:
      - postgres

  # Prometheus for Metrics Collection
  prometheus:
    image: prom/prometheus:latest
    container_name: deltavision-prometheus
    command:
      - '--config.file=/etc/prometheus/prometheus.yml'
      - '--storage.tsdb.path=/prometheus'
      - '--web.console.libraries=/etc/prometheus/console_libraries'
      - '--web.console.templates=/etc/prometheus/consoles'
      - '--storage.tsdb.retention.time=200h'
      - '--web.enable-lifecycle'
    volumes:
      - ./configs/prometheus.yml:/etc/prometheus/prometheus.yml:ro
      - prometheus_data:/prometheus
    ports:
      - "9090:9090"
    networks:
      - deltavision-network
    restart: unless-stopped

  # Node Exporter for System Metrics
  node-exporter:
    image: prom/node-exporter:latest
    container_name: deltavision-node-exporter
    command:
      - '--path.procfs=/host/proc'
      - '--path.rootfs=/rootfs'
      - '--path.sysfs=/host/sys'
      - '--collector.filesystem.mount-points-exclude=^/(sys|proc|dev|host|etc)($$|/)'
    volumes:
      - /proc:/host/proc:ro
      - /sys:/host/sys:ro
      - /:/rootfs:ro
    ports:
      - "9100:9100"
    networks:
      - deltavision-network
    restart: unless-stopped

volumes:
  postgres_data:
    driver: local
  pgadmin_data:
    driver: local
  redis_data:
    driver: local
  grafana_data:
    driver: local
  prometheus_data:
    driver: local

networks:
  deltavision-network:
    driver: bridge
    ipam:
      config:
        - subnet: 172.20.0.0/16
